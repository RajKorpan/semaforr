00050 // A persistent write function.
00051 static ssize_t write_r(int fd, const void *buf, size_t count, size_t ms)
00052 {
00053         double tMax = ms / 1000.0; // convert to seconds
00054         PosixTimer t;
00055         int r;
00056         unsigned int written = 0;
00057         while (written < count) {
00058                 r = write(fd, ((const unsigned char *)buf) + written, count - written);
00059                 // did something seriously go wrong?
00060                 if (r < 0 && errno != EINTR) {
00061                         __dbg(string("PosixSerial: failed to write: ") + string(strerror(errno)));
00062                         throw PosixSerial::WriteFailure();
00063                 } else if (r > 0) {
00064                         // progress!
00065                         written += r;
00066                         t.start(); // restart the timer
00067                 }
00068                 // are we out of time?
00069                 if (ms > 0 && t.elapsed() > tMax) {
00070                         __dbg(string("PosixSerial: write timed out"));
00071                         throw PosixSerial::WriteTimeout();
00072                 }
00073         }
00074         return written;

00125 void PosixSerial::connect()
00126 {
00127         if (!mFunctional) {
00128                 // Intercept I/O events on the port.
00129                 struct sigaction sa;
00130                 if (sigaction(SIGIO, 0, &sa) < 0) {
00131                         __dbg(string("PosixSerial: failed to retrieve sigaction"));
00132                         throw ConnectionFailure();
00133                 }
00134                 sa.sa_handler = __sighandler;
00135                 if (sigaction(SIGIO, &sa, 0) < 0) {
00136                         __dbg(string("PosixSerial: failed to override sigaction"));
00137                         throw ConnectionFailure();
00138                 }
00139 
00140                 // Open the port.
00141                 if ((mDevFD = open(mDevName.c_str(), O_RDWR | O_NOCTTY | O_NONBLOCK)) < 0) {
00142                         __dbg(string("PosixSerial: failed to open port ") + mDevName);
00143                         throw ConnectionFailure();
00144                 }
00145 
00146                 // Redirect all signals for the port to the current process.
00147                 if (fcntl(mDevFD, F_SETOWN, getpid()) < 0) {
00148                         __dbg(string("PosixSerial: failed to redirect signals: ") + string(strerror(errno)));
00149                         throw ConnectionFailure();
00150                 }
00151 
00152                 // Enable asynchronous I/O for the port.
00153                 if (fcntl(mDevFD, F_SETFL, FASYNC) < 0) {
00154                         __dbg(string("PosixSerial: failed to enable asynchronous I/O: ") + string(strerror(errno)));
00155                         throw ConnectionFailure();
00156                 }
00157 
00158                 // Get the port's current attributes.
00159                 struct termios attr;
00160                 memset((void *)&attr, 0, sizeof(attr));
00161                 if (tcgetattr(mDevFD, &attr) < 0 ) {
00162                         __dbg(string("PosixSerial: failed to acquire port attributes: ") + string(strerror(errno)));
00163                         throw ConnectionFailure();
00164                 }
00165 
00166                 // Set our attributes.
00167                 // TODO: attributes should probably be parameterized; just assume "raw" mode for now.
00168                 cfmakeraw(&attr);
00169                 cfsetspeed(&attr, mLineSpeed);
00170                 attr.c_cc[VTIME] = 0; // timeout in tenths of a second
00171                 attr.c_cc[VMIN] = 0;  // allow read to return 0
00172                 if (tcsetattr(mDevFD, TCSAFLUSH, &attr) < 0) {
00173                         __dbg(string("PosixSerial: failed set port attributes: ") + string(strerror(errno)));
00174                         throw ConnectionFailure();
